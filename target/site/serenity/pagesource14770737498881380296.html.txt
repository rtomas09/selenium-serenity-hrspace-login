function doLoginRedirectIfNeeded() {
  var KEY = 'logoutURLForLWBL',
      url = null,
      storage;
  try {
    if (storage = window.sessionStorage) {
      url = storage.getItem(KEY);
      if (url) {
        if (url == 'null') {
          url = null;
        }
        storage.removeItem(KEY);
      }
    }
  }
  catch (e) {
  }
  if (url) {
    window.location = url;
  }
  return !!url;
}
